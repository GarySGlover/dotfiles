  {
    config,
    lib,
    ...
  }:
  let
    theme = config.wolf.theme;
    faces = theme.faces;
  in
  {
    config = lib.mkIf config.wolf.roles.desktop {
      programs.waybar = {
        enable = true;
        systemd.enable = false;
        settings = {
          mainbar = {
            layer = "top";
            position = "top";
            margin = "${toString theme.gaps}";
            exclusive = "true";
            modules-left = [
              "hyprland/workspaces"
              "hyprland/submap"
            ];
            modules-center = [
              "hyprland/window"
            ];
            modules-right = [
              "pulseaudio"
              "battery"
              "clock"
            ];
            "hyprland/workspaces" = { };
            "hyprland/submap" = {
              always-on = "true";
            };
            "clock" = {
              format = "{:%Y%m%d-%H%M}";
              tooltip = "false";
            };
            pulseaudio = {
              format = "{desc} {volume}% {icon}";
              format-muted = "ï€¦";
            };
          };
        };
        style = ''
          <<waybar-style>>
        '';
      };
    };
  }
